## 5. 왜 추가 수정이 필요한가

현재 `access-gateway`는 검색기 화면을 앞에서 proxy하고 있다.

하지만 “학생용 결과 생성” 버튼은 검색기 내부 JS에서 기존 Worker로 직접 요청할 가능성이 높다.

기존 생성 Worker:

```text
https://curly-base-a1a9.koreapoorboy.workers.dev/generate
```

이 구조에서는 access-gateway가 버튼 클릭을 알 수 없다.

따라서 수정 방향은 아래와 같다.

```text
학생용 결과 생성 버튼
↓
/__mini/generate 로 요청
↓
access-gateway가 요청을 먼저 받음
↓
access-gateway가 접속 코드 확인
↓
만료일 확인
↓
남은 횟수 확인
↓
정상이면 1회 차감
↓
curly-base-a1a9 /generate 로 전달
↓
생성 결과 반환
```

```javascript
const TARGET_HOST = 'https://univ-search.pages.dev';
const TARGET_BASE_PATH = '/keyword-engine';

const GENERATE_WORKER_URL = 'https://curly-base-a1a9.koreapoorboy.workers.dev/generate';
const GENERATE_GATEWAY_PATH = '/__mini/generate';

const ACCESS_CODES = {
  'test-0508': {
    name: '테스트 접속권',
    expiresAt: '2026-05-31T23:59:59+09:00',
    maxUses: 100,
    enabled: true,
  },

  'expired-0508': {
    name: '만료 테스트 접속권',
    expiresAt: '2024-01-01T23:59:59+09:00',
    maxUses: 100,
    enabled: true,
  },
};

const COOKIE_NAME = 'univ_access_code';

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);
    const pathParts = url.pathname.split('/').filter(Boolean);
    const firstPart = pathParts[0] || '';

    if (request.method === 'OPTIONS') {
      return withCors(new Response(null, { status: 204 }));
    }

    if (url.pathname === '/health') {
      return json({
        ok: true,
        service: 'access-gateway',
        mode: 'generate-count-only',
        targetHost: TARGET_HOST,
        targetBasePath: TARGET_BASE_PATH,
        generateEndpoint: GENERATE_GATEWAY_PATH,
        hasKV: Boolean(env.ACCESS_KV),
      });
    }

    // 학생용 결과 생성 요청은 여기에서만 횟수 차감
    // 검색기 JS에서 /__mini/generate 로 요청하도록 바꾸면 이 분기로 들어온다.
    if (url.pathname === GENERATE_GATEWAY_PATH) {
      return handleGenerateRequest(request, env);
    }

    // 혹시 상대경로 때문에 /test-0508/__mini/generate 형태로 들어오는 경우도 처리
    if (pathParts.length >= 2 && pathParts[1] === '__mini' && pathParts[2] === 'generate') {
      return handleGenerateRequest(request, env, firstPart);
    }

    // 접속 코드로 검색기 입장
    // 예: /test-0508
    // 이때는 횟수 차감하지 않는다.
    if (ACCESS_CODES[firstPart]) {
      const check = await checkCode(firstPart, env, {
        checkUsageOnly: true,
      });

      if (!check.ok) {
        return deniedPage(check.message);
      }

      const strippedPath = '/' + pathParts.slice(1).join('/');
      const targetUrl = buildTargetUrl(strippedPath, url.search);

      const response = await proxyRequest(request, targetUrl);
      const newResponse = new Response(response.body, response);

      newResponse.headers.append(
        'Set-Cookie',
        `${COOKIE_NAME}=${firstPart}; Path=/; HttpOnly; Secure; SameSite=Lax; Max-Age=86400`
      );

      return newResponse;
    }

    // 이미 접속 코드를 통과한 뒤 내부 파일 요청
    const cookieCode = getCookie(request, COOKIE_NAME);

    if (!cookieCode || !ACCESS_CODES[cookieCode]) {
      return deniedPage('유효한 접속 코드가 없습니다.');
    }

    const check = await checkCode(cookieCode, env, {
      checkUsageOnly: true,
    });

    if (!check.ok) {
      return deniedPage(check.message);
    }

    const targetUrl = buildTargetUrl(url.pathname, url.search);
    return proxyRequest(request, targetUrl);
  },
};

async function handleGenerateRequest(request, env, pathCode = '') {
  if (request.method !== 'POST') {
    return json({
      ok: false,
      error: '학생용 결과 생성은 POST 요청만 허용됩니다.',
    }, 405);
  }

  const cookieCode = getCookie(request, COOKIE_NAME);
  const code = pathCode || cookieCode;

  if (!code || !ACCESS_CODES[code]) {
    return json({
      ok: false,
      error: '유효한 접속 코드가 없습니다. 전용 접속 주소로 다시 접속해 주세요.',
    }, 403);
  }

  const check = await checkCode(code, env, {
    checkUsageOnly: false,
  });

  if (!check.ok) {
    return json({
      ok: false,
      error: check.message,
    }, 403);
  }

  const upstreamRequest = new Request(GENERATE_WORKER_URL, {
    method: 'POST',
    headers: request.headers,
    body: request.body,
    redirect: 'follow',
  });

  const upstreamResponse = await fetch(upstreamRequest);
  const responseHeaders = new Headers(upstreamResponse.headers);

  responseHeaders.set('Access-Control-Allow-Origin', '*');
  responseHeaders.set('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
  responseHeaders.set('Access-Control-Allow-Headers', 'Content-Type, Authorization');

  // 생성 Worker가 성공했을 때만 1회 차감
  if (upstreamResponse.ok) {
    await increaseUsage(code, env);
  }

  return new Response(upstreamResponse.body, {
    status: upstreamResponse.status,
    statusText: upstreamResponse.statusText,
    headers: responseHeaders,
  });
}

function buildTargetUrl(pathname, search = '') {
  let path = pathname || '/';

  if (!path.startsWith('/')) {
    path = '/' + path;
  }

  if (path === TARGET_BASE_PATH || path.startsWith(TARGET_BASE_PATH + '/')) {
    const targetUrl = new URL(TARGET_HOST + path);
    targetUrl.search = search || '';
    return targetUrl;
  }

  if (path === '/') {
    const targetUrl = new URL(TARGET_HOST + TARGET_BASE_PATH + '/');
    targetUrl.search = search || '';
    return targetUrl;
  }

  const targetUrl = new URL(TARGET_HOST + TARGET_BASE_PATH + path);
  targetUrl.search = search || '';
  return targetUrl;
}

async function checkCode(code, env, options = {}) {
  const item = ACCESS_CODES[code];

  if (!item) {
    return {
      ok: false,
      message: '존재하지 않는 접속 코드입니다.',
    };
  }

  if (!item.enabled) {
    return {
      ok: false,
      message: '사용이 중지된 접속 코드입니다.',
    };
  }

  const now = new Date();
  const expiresAt = new Date(item.expiresAt);

  if (Number.isNaN(expiresAt.getTime())) {
    return {
      ok: false,
      message: '접속 코드의 만료일 설정이 잘못되었습니다.',
    };
  }

  if (now > expiresAt) {
    return {
      ok: false,
      message: '접속 기간이 만료되었습니다. 관리자에게 문의해 주세요.',
    };
  }

  if (item.maxUses && !env.ACCESS_KV) {
    return {
      ok: false,
      message: '사용 횟수 저장소가 연결되지 않았습니다. 관리자에게 문의해 주세요.',
    };
  }

  if (item.maxUses && env.ACCESS_KV) {
    const currentUses = await getUsage(code, env);

    if (currentUses >= item.maxUses) {
      return {
        ok: false,
        message: `접속 가능 횟수 ${item.maxUses}회를 모두 사용했습니다. 관리자에게 문의해 주세요.`,
      };
    }
  }

  return {
    ok: true,
  };
}

async function getUsage(code, env) {
  if (!env.ACCESS_KV) return 0;
  const usageKey = `access:${code}:uses`;
  const currentUsesRaw = await env.ACCESS_KV.get(usageKey);
  return Number(currentUsesRaw || 0);
}

async function increaseUsage(code, env) {
  const item = ACCESS_CODES[code];
  if (!item?.maxUses || !env.ACCESS_KV) return;

  const usageKey = `access:${code}:uses`;
  const currentUses = await getUsage(code, env);
  await env.ACCESS_KV.put(usageKey, String(currentUses + 1));
}

async function proxyRequest(request, targetUrl) {
  const headers = new Headers(request.headers);

  headers.set('Host', new URL(TARGET_HOST).host);
  headers.delete('cf-connecting-ip');
  headers.delete('cf-ipcountry');
  headers.delete('cf-ray');
  headers.delete('x-forwarded-proto');
  headers.delete('x-real-ip');

  const newRequest = new Request(targetUrl.toString(), {
    method: request.method,
    headers,
    body:
      request.method === 'GET' || request.method === 'HEAD'
        ? undefined
        : request.body,
    redirect: 'follow',
  });

  const response = await fetch(newRequest);
  const responseHeaders = new Headers(response.headers);

  responseHeaders.delete('content-security-policy');
  responseHeaders.delete('content-security-policy-report-only');
  responseHeaders.delete('x-frame-options');

  return new Response(response.body, {
    status: response.status,
    statusText: response.statusText,
    headers: responseHeaders,
  });
}

function getCookie(request, name) {
  const cookie = request.headers.get('Cookie') || '';
  const parts = cookie.split(';').map((v) => v.trim());

  for (const part of parts) {
    const [key, ...rest] = part.split('=');
    if (key === name) {
      return rest.join('=');
    }
  }

  return '';
}
```

# NEXT_STEP_2. keyword-engine ZIP 내부 JS 패치 필요

access-gateway만 바꿔서는 아직 완성되지 않는다.

검색기 내부 JS가 학생용 결과 생성 요청을 기존 Worker로 직접 보내고 있으면,
access-gateway가 버튼 클릭을 알 수 없다.

따라서 `keyword-engine.zip` 안에서 아래 파일을 수정해야 한다.

우선 확인해야 할 파일 후보:

```text
keyword-engine/assets/js/mini_worker_generate_bridge_v32.js
```

이 파일 안에서 기존 생성 Worker 주소를 찾는다.

찾을 가능성이 높은 문자열:

```text
curly-base-a1a9.koreapoorboy.workers.dev/generate
```

또는:

```text
/generate
```

또는:

```text
MINI_WORKER
```

또는:

```text
generate
```

수정 목표:

기존처럼 외부 Worker로 직접 보내지 말고,
현재 접속 중인 access-gateway 내부 경로로 보내야 한다.

변경할 목표 주소:

```text
/__mini/generate
```

즉 JS 내부 생성 요청 endpoint를 아래처럼 바꿔야 한다.

```javascript
const MINI_GENERATE_ENDPOINT = '/__mini/generate';
```

또는 기존 fetch가 아래처럼 되어 있다면:

```javascript
fetch('https://curly-base-a1a9.koreapoorboy.workers.dev/generate', ...)
```

아래처럼 바꾼다.

```javascript
fetch('/__mini/generate', ...)
```

이렇게 해야 가상주소 접속 상태에서 학생용 결과 생성 버튼을 누를 때:

```text

→ 만료일 확인
→ 사용 횟수 확인
→ 1회 차감
→ curly-base-a1a9 /generate로 전달
```

구조가 된다.
